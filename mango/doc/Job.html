def initialize h={}
  if h[:id]
    @id = h[:id]
  else
    @id = gen_id
    self.state.value = h.delete(:state) || "NEW"
    self.from.value = h.delete(:from)
    if h[:to]
      self.to.value = h.delete(:to)
    elsif h[:queue]
      self.to.value = h.delete(:queue)
      queue(self.to.value, @id)
    else
      queue << @id
    end
    h.each_pair {|k,v| self.attrs[k] = v }
  end
end

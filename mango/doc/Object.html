def gen_id *i
  if i[0]
    n = i[0].to_i
  else
    n = 6
  end
  @a = []
  until @a.join('') != '' && !Redis::HashKey.new("JOBS").has_key?(@a.join(''))
    n.times {@a << rand(9)}
  end
  return @a.join('')
end

def listen f
  @learning[f] = Process.detach( fork {
                    t = Tw::Client::Stream.new
                    t.filter(f, language: 'english') { |m| stats[:main].tag(m.text.downcase) }
                  }
                )
end

def queue *q
  if q[0]
    if q[1]
      Redis::List.new("queue:#{q[0]}") << q[1]
    else
      Redis::List.new("queue:#{q[0]}")
    end
  else
    Redis::List.new("queue:main")
  end
end

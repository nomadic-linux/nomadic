def tag str
  s = @sentiment.sentiment str
  w = str.gsub('.', '').gsub(',','').gsub('?','').gsub('!','').split(' ')
  w.each { |e|
    if s == :positive
      @feeling.incr(e)
    elsif s == :negative
      @feeling.decr(e)
    end
    w.each { |ee|
             if ee != e;
               add(e, ee);
               if s == :positive
                 @occour.incr(e + ":" + ee)
                 @occour.incr(ee + ":" + e)
               elsif s == :negative
                 @occour.decr(e + ":" + ee)
                 @occour.decr(ee + ":" + e)
               end
             end
           }
  }
end
